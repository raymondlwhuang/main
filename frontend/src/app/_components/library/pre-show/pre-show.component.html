Selectors

  • document.getElementById("demo"); //Get the element with the specified ID:
  • document.getElementsByClassName("example"); //Get all elements with the specified class name
  • document.getElementsByName("fname"); //Get all elements with the specified name
  • document.getElementsByTagName("LI"); //Get all elements in the document with the specified tag name
  • document.querySelector(".example"); //Get the first element in the document with class="example"
  • document.querySelectorAll(".example"); //Get all elements in the document with class="example"
